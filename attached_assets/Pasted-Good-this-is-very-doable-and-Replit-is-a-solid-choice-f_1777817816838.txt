CREATE TABLE IF NOT EXISTS rates (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    origin TEXT,
    destination TEXT,
    equipment TEXT,
    rate REAL,
    rate_per_mile REAL,
    pickup_date TEXT,
    source TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

🧠 4. Database Helper (db.py)

import sqlite3
DB_NAME = "freight.db"
def get_conn():
    return sqlite3.connect(DB_NAME)
def init_db():
    conn = get_conn()
    with open("schema.sql", "r") as f:
        conn.executescript(f.read())
    conn.close()
def insert_rate(data):
    conn = get_conn()
    cursor = conn.cursor()
    cursor.execute("""
        INSERT INTO rates (origin, destination, equipment, rate, rate_per_mile, pickup_date, source)
        VALUES (?, ?, ?, ?, ?, ?, ?)
    """, data)
    conn.commit()
    conn.close()
def get_rates():
    conn = get_conn()
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM rates ORDER BY created_at DESC LIMIT 50")
    rows = cursor.fetchall()
    conn.close()
    return rows

🕷️ 5. Scraper (scraper.py)

⚠️ This uses a generic structure—you’ll tweak selectors depending on the site.

from playwright.sync_api import sync_playwright
from db import insert_rate
def scrape():
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        page = browser.new_page()
        # Example free loadboard (replace if needed)
        page.goto("https://tbosloadboard.com/")
        page.wait_for_timeout(5000)
        rows = page.query_selector_all("table tr")
        for row in rows[1:10]:  # limit for demo
            cols = row.query_selector_all("td")
            if len(cols) < 5:
                continue
            origin = cols[0].inner_text()
            destination = cols[1].inner_text()
            equipment = cols[2].inner_text()
            rate = cols[3].inner_text().replace("$", "")
            rpm = cols[4].inner_text()
            data = (
                origin,
                destination,
                equipment,
                float(rate) if rate else 0,
                float(rpm) if rpm else 0,
                "N/A",
                "TBOS"
            )
            insert_rate(data)
        browser.close()

🌐 6. API Server (main.py)

from flask import Flask, jsonify
from db import init_db, get_rates
from scraper import scrape
app = Flask(__name__)
@app.route("/")
def home():
    return "Freight Tracker Running"
@app.route("/scrape")
def run_scrape():
    scrape()
    return "Scraping complete"
@app.route("/rates")
def rates():
    data = get_rates()
    return jsonify(data)
if __name__ == "__main__":
    init_db()
    app.run(host="0.0.0.0", port=3000)
